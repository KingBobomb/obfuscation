 69    def __initialize_graph(self):
 70        # Method to create an initial graph of the map separate from
 71        # the actual nodes so the AI doesn't instantly know about
 72        # unblocked doors
 73
 74        # Queue for storing new nodes to visit.
 75        node_queue = deque([self.__curr_loc])
 76        # Set of already visited Nodes.
 77        visited_nodes = {self.__curr_loc}
 78        # Initialize the AI's graph with the first location and its exits.
 79        self.__graph = {self.__curr_loc: self.__curr_loc.get_exits()}
 80        while node_queue:
 81            curr_node = node_queue.popleft()
 82
 83            # Calculate the number of interactions in a room.
 84            interactions = (len(curr_node.get_npcs())
 85                            + len(curr_node.get_items()))
 86            self.__num_interactions[curr_node] = interactions
 87
 88            for node in curr_node.get_exits().keys():
 89                if node not in visited_nodes:
 90                    node_queue.append(node)
 91
 92            visited_nodes.add(curr_node)
 93            self.__graph[curr_node] = curr_node.get_exits()

 95    def __get_path_to(self, target_loc):
 96        # Method to get a pathway to a desired node in the AI's graph using BFS
 97        # Unlike the initialize graph function, this runs on the AI's graph
 98
 99        # Keeps track of the nodes to visit
100        node_queue = deque([self.__curr_loc])
101        # Keeps track of visited nodes
102        visited_nodes = [self.__curr_loc]
103        # Keeps track of the path we took to reach a node
104        predecessor_node = {self.__curr_loc: None}
105
106        # Run until there aren't any nodes to visit
107        while node_queue:
108            # Remove the first element of our queue
109            curr_node = node_queue.popleft()
110
111            # If we've reached our target destination, return the path back to our starting point
112            if curr_node == target_loc:
113                final_path = []
114                while predecessor_node[curr_node] is not None:
115                    final_path.append(curr_node)
116                    curr_node = predecessor_node[curr_node]
117                return final_path
118
119            # Add new nodes to the queue and mark the current node as its predecessor
120            curr_exits = self.__graph[curr_node]
121            for node in curr_exits.keys():
122                if node not in visited_nodes and curr_exits[node] != 0:
123                    node_queue.append(node)
124                    if node in predecessor_node:
125                        predecessor_node[node] = curr_node
126
127            visited_nodes.append(curr_node)
128
129        # If no path is found to the given destination, return None
130        return None

186        # Function to update the AI's knowledge of the room it's in.
187
188        # Clear the current list of NPCs in the AI's current room.
189        self.__npcs_in_room = []
190        # Clear the list of items the AI hasn't found in the current room.

192
193        # Add NPCs to our list of NPCs to talk to if we haven't spoken to them yet.

195            if npc not in self.__npcs_spoken_to:
196                self.__npcs_in_room.append(npc)
197
198        # Add items to our list of items to search for if we haven't found to them yet.
199        for new_item in self.__curr_loc.get_items():
200            if new_item not in self.__items_found:
201                self.__items_in_room.append(new_item)
202
203        # Update the remaining number of interactions at this location.
204        interactions = len(self.__items_in_room) + len(self.__npcs_in_room)
205        self.__num_interactions[self.__curr_loc] = interactions
206
207        room_exits = self.__curr_loc.get_exits()

210        if room_exits != self.__graph[self.__curr_loc]:
211            # Update graph to match findings.
212            for exits in room_exits.keys():
213                self.__graph[self.__curr_loc][exits] = room_exits[exits]

215            # Attempt to path to any items we couldn't before.

217                if self.__incriminating_item_found_dict[incrim_item] != 1:

221    def take_turn(self):
222        """Executes on turn cycle for the AI Agent and determines if the game should end.
223
224        During its turn the AI will update it's knowledge of the room it's currently in,
225        decide what action to take (between moving, talking to an NPC, or searching for an item),
226        execute those actions, and check if the conditions are met to end the game.
227
228        Returns:
229            bool : True if the game should end, false if the game should continue
230        """
231        # Update the Ai's knowledge about the room its in.
232        self.__update_room_knowledge()
233        # Have the AI make a choice on what to do.
234        self.__make_choice()
235
236        # If the suspicion meter exceeds 100, end the game by returning true.
237        if self.__suspicion_meter >= 100:
238            return True
239
240        # If the AI has found every incriminating item, end the game by returning true
241        if 0 not in self.__incriminating_item_found_dict.values():
242            return True
243
244        # If neither are true, return false to continue the game.
245        return False

255    def __handle_priority_behavior(self, item, loc):
256        # Helper for the __make_choice function that checks and executes priority behaviors for
257        # a given item. Returns true if a priority behavior was executed and false if not
258
259        # Check if we know where an item is.
260        if loc is None:
261            return False
262
263        # Check that we haven't already found the current item.
264        if self.__incriminating_item_found_dict[item] == 1:
265            return False
266
267        # Check if we are in the same room as an incriminating item. If so, trigger a helper.
268        if self.__curr_loc == loc:
269            return self.__handle_incriminating_in_room(item)
270
271        # Check if a pathway exists to our item. If so continue in that direction.
272        if self.__incriminating_item_path_dict[item]:
273            self.__move_self(self.__incriminating_item_path_dict[item][-1])
274            self.__incriminating_item_path_dict[item].pop()
275            return True
276
277        return False

279    def __handle_incriminating_in_room(self, item):
280        # Helper for the __handle_priority_behavior function for when incriminating items are in
281        # the same room as the AI
282
283        # Check if there are any items left to search for in the current room.
284        if not self.__items_in_room:
285            # If an item should be here but isn't, we assume the player took it.
286            # So we destroy the path and location of the item.
287            self.__incriminating_item_loc_dict[item] = None
288            self.__incriminating_item_path_dict[item] = None
289            return False
290
291        # Randomly search for an item in the room to emulate searching furniture
292        to_search = choices(self.__items_in_room, k=1)[0]
293        self.__search_room(to_search)
294        return True

296    def __make_choice(self):
297        # Helper method to manage the logic for the AI's turn.
298
299        move_mult = 1
300        search_mult = 1
301        talk_mult = 1
302
303        # See if any priority behaviors need to occur for key items
304        for incrim_item, loc in self.__incriminating_item_loc_dict.items():
305            if self.__handle_priority_behavior(incrim_item, loc):
306                return
307
308        # If no predefined behavior was selected, adjust our weights for a
309        # random selection
310        # Note: Investigative style is intended to influence the move mult in
311        # a later release
312
313        # Talk mult is proportional to number of NPCs left to talk to in room.
314        talk_mult = len(self.__npcs_in_room)
315        # Search mult is proportional to number of items left in room.
316        search_mult = len(self.__items_in_room)
317
318        choice_weights = [self.__base_weights['move'] * move_mult,
319                          self.__base_weights['search'] * search_mult,
320                          self.__base_weights['talk'] * talk_mult]
321
322        chosen = choices(list(self.__base_weights.keys()),
323                         weights=choice_weights, k=1)[0]
324
325        if chosen == 'move':
326            # Weights are used to give a preference to moving towards
327            # places with more interactions and not moving back to the
328            # previous location.
329            # Note: Like the graph, this is based on what the AI believes
330            # about each location.
331            room_weight_dict = {}
332
333            # Add the number of interactions for each open exit
334            for loc, block in self.__graph[self.__curr_loc].items():
335                if loc in self.__prev_3_loc:
336                    room_weight_dict[loc] = self.__num_interactions[loc] + 1
337                elif block == 1:
338                    room_weight_dict[loc] = self.__num_interactions[loc] + 4
339                else:
340                    room_weight_dict[loc] = 0
341            if room_weight_dict:
342                to_move = choices(list(room_weight_dict.keys()),
343                                  weights=list(room_weight_dict.values()), k=1
344                                  )[0]
345                self.__move_self(to_move)
346            else:
347                print("The AI has no valid move to make.")
348        elif chosen == 'search':
349            # Randomly search for an item to emulate searching furniture
350            to_search = choices(self.__items_in_room, k=1)[0]
351            self.__search_room(to_search)
352        else:
353            # Randomly select an NPC in the room we haven't talked to
354            to_talk = choices(self.__npcs_in_room, k=1)[0]
355            self.__talk_to_npc(to_talk)
356
357        return

298        # Helper method to manage the logic for the AI's turn.
299
300        move_mult = 1
301        search_mult = 1
302        talk_mult = 1
303
304        # See if any priority behaviors need to occur for key items
305        for incrim_item, loc in self.__incriminating_item_loc_dict.items():
306            if self.__handle_priority_behavior(incrim_item, loc):
307                return
308
309        # If no predefined behavior was selected, adjust our weights for a
310        # random selection
311        # Note: Investigative style is intended to influence the move mult in
312        # a later release
313
314        # Talk mult is proportional to number of NPCs left to talk to in room.
315        talk_mult = len(self.__npcs_in_room)
316        # Search mult is proportional to number of items left in room.
317        search_mult = len(self.__items_in_room)
318
319        choice_weights = [self.__base_weights['move'] * move_mult,
320                          self.__base_weights['search'] * search_mult,
321                          self.__base_weights['talk'] * talk_mult]
322
323        chosen = choices(list(self.__base_weights.keys()),
324                         weights=choice_weights, k=1)[0]
325
326        if chosen == 'move':
327            # Weights are used to give a preference to moving towards
328            # places with more interactions and not moving back to the
329            # previous location.
330            # Note: Like the graph, this is based on what the AI believes
331            # about each location.
332            room_weight_dict = {}
333
334            # Add the number of interactions for each open exit
335            for loc, block in self.__graph[self.__curr_loc].items():
336                if loc in self.__prev_3_loc:

341                    room_weight_dict[loc] = 0
342            if room_weight_dict:

355            to_talk = choices(self.__npcs_in_room, k=1)[0]
